### 1.3最短路

#### a.dijkstra

- 求非负权图的单源最短路。
  简单应用：记录路径，求最短路径数。

```c++
#include<iostream>
#include<vector>
#include<cstring>
#include<queue>
#include<stack>
#define pii pair<int,int>
using namespace std;
const int inf = 1e9;
const int mod = 1e5 + 3;
const int N = 1e6 + 5;

int sum[N];//记录最短路径数
int pre[N];//记录前驱

int n, m, s;
vector<pii>g[N << 1];

int dis[N], vis[N];
vector<int>L[N];
priority_queue<pii, vector<pii>, greater<pii>>Q; //fs是w，se是u
void dij() {
	for (int i = 1; i <= n; i++)dis[i] = inf;
	dis[s] = 0;

	for (int i = 1; i <= n; i++)pre[i] = -1;//前驱标记为-1
	sum[s] = 1;

	Q.push({ 0, s });
	while (!Q.empty()) {
		pair<int, int>x = Q.top();
		Q.pop();
		int u = x.second;

		if (vis[u])continue;
		vis[u] = 1;

		for (auto to : g[u]) {
			int v = to.first, w = to.second;//fs是u，se是w
			if (dis[v] > dis[u] + w) {
				dis[v] = dis[u] + w;
				Q.push({ dis[v], v });

				sum[v] = sum[u];
				pre[v] = u;
			} else if (dis[v] == dis[u] + w)
				sum[v] = (sum[v] + sum[u]) % mod;
		}
	}
}
void path(int u) {
	//用栈逆序输出，或者递归输出
	stack<int>s;
	do {
		s.push(u);
		u = pre[u];
	} while (u != -1);
	while (!s.empty()) {
		cout << s.top() << ' ';
		s.pop();
	}
	cout << '\n';
}

int main() {
	cin >> n >> m >> s;
	for (int i = 1; i <= m; i++) {
		int u, v, w;
		cin >> u >> v >> w;
		g[u].push_back({ v, w });
		g[v].push_back({ u, w });
	}
	dij();
	for (int i = 1; i <= n; i++)path(i);
	return 0;
}
```

#### b.floyd

- 求多源最短路。
  简单应用：图的传递闭包，记录最短路径。

```c++
#include<iostream>
#include<stack>
using namespace std;
const int N = 1e3 + 5;
const int inf = 1e9;
int n, m, s;
int dis[N][N];
int pre[N][N];
void path(int s, int t) {
	if (dis[s][t] == inf) {
		cout << "none";
		return;
	}
	stack<int>sk;
	do {
		sk.push(t);
		t = pre[s][t];
	} while (t != -1);
	sk.push(s);
	while (!sk.empty()) {
		cout << sk.top() << ' ';
		sk.pop();
	}
	cout << '\n';
}
int main() {
	cin >> n >> m >> s;
	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= n; j++)
			dis[i][j] = i == j ? 0 : inf,
			            pre[i][j] = -1;//初始化
	for (int i = 1; i <= m; i++) {
		int u, v, w;
		cin >> u >> v >> w;
		dis[u][v] = w;
	}
	for (int k = 1; k <= n; k++)
		for (int i = 1; i <= n; i++)
			for (int j = 1; j <= n; j++)
				if (dis[i][j] > dis[i][k] + dis[k][j])
					dis[i][j] = dis[i][k] + dis[k][j],
					            pre[i][j] = k;//记录前驱
	//dis[i][j] |= dis[i][k] & dis[k][j];//传递闭包
	path(1, 4);
	return 0;
}
```

### 1.4树上问题

#### a.拓扑序数

- 求**树**或**森林**的拓扑序数公式：

$$
dp[u]=(\prod dp[v])*\frac{(sz[u]-1)!}{\prod (sz[v]!)}=(sz[u]-1)!*\prod\frac{dp[v]}{sz[v]!}
$$

$$
n!*\prod_{i=1}^{i=n}{\frac{1}{sz[i]}}
$$

#### b.LCA

- 基于倍增

```c++
int n;
vector<int>g[N];
///LCA
int dep[N], fa[N][22];
void dfs(int u, int pre) {
	//预处理深度信息和树上st表
	dep[u] = dep[pre] + 1;
	fa[u][0] = pre;//当前节点的第2^0=1个节点是它的父亲节点
	//当前节点的第2^i个节点是它的第2^i-1个节点的第2^i-1个节点
	for (int i = 1; (1 << i) <= dep[u]; i++)
		fa[u][i] = fa[fa[u][i - 1]][i - 1];
	for (auto v : g[u])
		if (v != pre)//无向边
			dfs(v, u);
}
int LCA(int x, int y) {
	if (dep[x] < dep[y])swap(x, y);//保证x的深度大于y的深度
	for (int i = 20; i >= 0; i--) {
		//x，y跳到同一深度
		if (dep[fa[x][i]] >= dep[y])
			x = fa[x][i];
		if (x == y)return x;//特判
	}
	for (int i = 20; i >= 0; i--)//同时跳
		if (fa[x][i] != fa[y][i])
			x = fa[x][i], y = fa[y][i];
	return fa[x][0];//最后LCA是它们的父亲节点
}
int main() {
	cin >> n;
	for (int i = 1; i <= n - 1; i++) {
		int u, v;
		cin >> u >> v;
		g[u].push_back(v);
		g[v].push_back(u);
	}
	dfs(1, 0);
	return 0;
}
```

#### c.树上差分

- 疑惑：在差分数组上求和或在新数组上求和对答案的影响。

- 树上差分又分为**点差分**和**边差分**：

  - 点差分：
    - 路径差分：
      对于路径$u$到$v$的点权值加$x$，对差分数组的操作为：$diff[u]+=x,diff[v]+=x,diff[o]-=x,diff[p]-=x$，其中，o为$u$和$v$的LCA，$p$为$o$的父节点。对于LCA是根节点的特殊情况，相当于只修改$u$节点的差分数组。最后**从下往上**dfs求前缀和数组即可。
    - 子树差分：
      对以$u$为根节点的子树权值全部加$x$，对差分数组的操作为：$sdiff[u]+=x$。最后**从上往下**dfs求前缀和数组即可。
    - 题意：对一棵完全二叉树给定$4$个操作，操作$1$为将以$x$为根节点的子树所有点权值加$1$，操作$3$为将$x$到根节点路径上的点权值加1。

  ```c++
  #include<iostream>
  using namespace std;
  const int N = 1e7 + 5;
  int n, m;
  int s[N], r[N], g;//分别为子树差分和路径差分//g是对整棵树操作
  int ssum[N], rsum[N];
  int ans[N];
  
  void dfs(int k, int fa) {
  	if (k > n)return;
  	ssum[k] = s[k] + ssum[fa];
  	dfs(k + k, k);
  	dfs(k + k + 1, k);
  	rsum[k] = r[k] + rsum[k + k] + rsum[k + k + 1];
  }
  int main() {
  	cin >> n >> m;
  	for (int i = 1; i <= m; i++) {
  		int op, x;
  		cin >> op >> x;
  		if (op == 1)
  			s[x]++;
  		else if (op == 2)
  			s[x]--, g++;
  		else if (op == 3)
  			r[x]++;
  		else
  			r[x]--, g++;
  	}
  	dfs(1, -1);
  	for (int i = 1; i <= n; i++)ans[ssum[i] + rsum[i] + g]++;
  	for (int i = 0; i <= m; i++)cout << ans[i] << ' ';
  	return 0;
  }
  ```
  
  

  - 边差分:
    将路径$u$到$u$的边的权值加$x$，对差分数组的操作为：$diff[u]+=x,diff[v]+=x,diff[o]-=2x$，其中$o$为$u$和$v$的LCA。最后**从下往上**dfs求差分数组的前缀和，$diff[v]$的值即为$u$到$v$所在边的权值变化量。
  - 题意：求边被覆盖的次数，按边的输入顺序输出。

  ```c++
  #include<iostream>
  #include<vector>
  using namespace std;
  const int N = 2e5 + 5;
  int n, k;
  int u[N], v[N];
  vector<int>g[N];
  //LCA
  int dep[N], fa[N][22], son[N];
  int diff[N];
  void get(int u, int fa) {
  	for (int v : g[u]) {
  		if (v == fa)continue;
  		get(v, u);
  		diff[u] += diff[v];
  	}
  }
  int main() {
  	cin >> n;
  	for (int i = 1; i <= n - 1; i++) {
  		cin >> u[i] >> v[i];
  		g[u[i]].push_back(v[i]), g[v[i]].push_back(u[i]);
  	}
  	dfs(1, -1);
  	cin >> k;
  	for (int i = 1; i <= k; i++) {
  		int u, v;
  		cin >> u >> v;
  		int o = lca(u, v);
  		diff[u]++, diff[v]++;
  		diff[o] -= 2;
  	}
  	get(1, -1);
  	for (int i = 1; i <= n - 1; i++) {
  		//根据儿子大小判断
  		if (son[u[i]] < son[v[i]])cout << diff[u[i]] << ' ';
  		else cout << diff[v[i]] << ' ';
  	}
  	return 0;
  }
  ```

  ```c++
  int n, m;

  int vis[N], dis[N], cnt;
  int prim() {
  	int ans = 0;
  
  	for (int i = 1; i <= n; i++)dis[i] = inf;
  	dis[1] = 0;
  
  	Q.push({ 0, 1 });
  	while (!Q.empty() && cnt < n) {
  		pii x = Q.top();
  		Q.pop();
  ```

  ```c++
  int n, m;
  struct edge { ///从定义出发存边
  	int a, b, w;
  	///重载比较权值
  	bool operator<(const edge& x)const {
  		return w < x.w;

  	for (int i = 1; i <= n; i++)fa[i] = i;
  ```

### 1.6强连通分量

- 缩点模板

```c++
#include<iostream>
#include<vector>
#include<queue>
using namespace std;
const int N = 1e5 + 5;
int n, m, w[N];
int u[N], v[N];
vector<int>g[N];///原图

int low[N], dfn[N], vis[N], cnt;///low是起始戳，dfn是时间戳
int st[N], top;///栈
int fa[N];///节点所在scc的编号
void tarjan(int u) {
	low[u] = dfn[u] = ++cnt;
	st[++top] = u, vis[u] = 1;
	for (int v : g[u]) {
		if (!dfn[v]) {
			tarjan(v);
			low[u] = min(low[u], low[v]);
		} else if (vis[v])
			low[u] = min(low[u], dfn[v]);
	}
	if (dfn[u] == low[u]) {
		int v;
		while (v = st[top--]) {
			fa[v] = u;///缩点，或者说染色
			vis[v] = 0;
			if (u == v)break;
			w[u] += w[v];///将一个强连通分量缩成一个点
		}
	}
}
vector<int>ng[N];///新图
int in[N], ans[N];///入度
int topo() {
	queue<int>Q;
	for (int i = 1; i <= n; i++)
		if (!in[i]) {
			Q.push(i);
			ans[i] = w[i];
		}
	while (!Q.empty()) {
		int u = Q.front();
		Q.pop();
		for (int v : ng[u]) {
			ans[v] = max(ans[v], ans[u] + w[v]);///dp
			if (--in[v] == 0)Q.push(v);
		}
	}
	int sum = 0;
	for (int i = 1; i <= n; i++)sum = max(sum, ans[i]);
	return sum;
}
int main() {
	cin >> n >> m;
	for (int i = 1; i <= n; i++)cin >> w[i];
	for (int i = 1; i <= m; i++) {
		cin >> u[i] >> v[i];///储存边
		g[u[i]].push_back(v[i]);///原图
	}
	for (int i = 1; i <= n; i++)///找强连通分量
		if (!dfn[i])tarjan(i);
	for (int i = 1; i <= m; i++) { ///建新图
		int U = fa[u[i]], V = fa[v[i]];
		if (U != V) {
			ng[U].push_back(V);
			in[V]++;
		}
	}
	cout << topo();
	return 0;
}
```

### 3.3线段树

- 区间修改，区间查询

```c++
#define int long long
int n, m, sum[N << 2], add[N << 2], a[N];
void bt(int k, int l, int r) {
	add[k] = 0;//建树时初始化标记
	if (l == r) {
		sum[k] = a[r];
		return;
	}
	int m = (l + r) >> 1;
	bt(k + k, l, m);
	bt(k + k + 1, m + 1, r);
	sum[k] = sum[k + k] + sum[k + k + 1];
}
void pushdown(int k, int l, int r) {
	int m = (l + r) >> 1;
	add[k + k] += add[k];
	sum[k + k] += (m - l + 1) * add[k];
	add[k + k + 1] += add[k];
	sum[k + k + 1] += (r - m) * add[k];
	add[k] = 0;
}
void update(int k, int l, int r, int x, int y, int z) {
	if (l == x && r == y) {
		add[k] += z;//打上标记的同时更新区间，保证当前区间的值是正确的
		sum[k] += (r - l + 1) * z;
		return;
	}
	pushdown(k, l, r);
	int m = (l + r) >> 1;
	if (y <= m)
		update(k + k, l, m, x, y, z);
	else if (x > m)
		update(k + k + 1, m + 1, r, x, y, z);
	else
		update(k + k, l, m, x, m, z),
		       update(k + k + 1, m + 1, r, m + 1, y, z);
	sum[k] = sum[k + k] + sum[k + k + 1];
}
int query(int k, int l, int r, int x, int y) {
	if (l == x && r == y)return sum[k];//直接返回
	pushdown(k, l, r);
	int m = (l + r) >> 1;
	int ans = 0;
	if (y <= m)
		ans = query(k + k, l, m, x, y);
	else if (x > m)
		ans = query(k + k + 1, m + 1, r, x, y);
	else
		ans = query(k + k, l, m, x, m) + query(k + k + 1, m + 1, r, m + 1, y);
	sum[k] = sum[k + k] + sum[k + k + 1];//需要pushup才能返回
	return ans;
}
signed main() {
	cin >> n >> m;
	for (int i = 1; i <= n; i++)cin >> a[i];
	bt(1, 1, n);
	while (m--) {
		int op, x, y, k;
		cin >> op;
		if (op == 1) {
			cin >> x >> y >> k;
			update(1, 1, n, x, y, k);
```

### 5.5简单莫队

- p2709
- 题目大意：有一个长为$n$的整数序列$a$，值域为$[1,k]$。一共有$m$个询问，每个询问给定一个区间$[l,r]$，求$\sum_{i=1}^{k}{c_i}$。其中$c_i$表示数字$i$在$[l,r]$中的出现次数。

```c++
const int N = 5e4 + 5;
int n, m, k;
long long ans[N];
int a[N], b[N];///a是原数组，b是桶,统计每个数出现的次数
int sz;///分块大小
struct node {
	int l, r, id;
	bool operator<(const node& x) {
		if (l / sz != x.l / sz)return l < x.l; //不在相同块按块从左往右排序
		return r < x.r;//在相同块则按r降序排序
	}
} mo[N];
long long c;///随着区间移动而更新的答案
void add(int i) {
	c += 2 * b[i] + 1;
	b[i]++;
}
void del(int i) {
	c -= 2 * b[i] - 1;
	b[i]--;
}
int  main() {
	cin >> n >> m >> k;///根本与k无关
	sz = sqrt(n);
	for (int i = 1; i <= n; i++)cin >> a[i];
	for (int i = 1; i <= m; i++) {
		cin >> mo[i].l >> mo[i].r;
		mo[i].id = i;
	}
	sort(mo + 1, mo + 1 + m);
	///l:上一次查询的左端点，r:上一次查询的右端点
	for (int i = 1, l = 1, r = 0; i <= m; i++) {
		///挪动操作
		while (l > mo[i].l)l--, add(a[l]);
		while (r < mo[i].r)r++, add(a[r]);
		while (l < mo[i].l)del(a[l]), l++;
		while (r > mo[i].r)del(a[r]), r--;
		ans[mo[i].id] = c;
	}
	for (int i = 1; i <= m; i++)cout << ans[i] << '\n';
	return 0;
}
```
